2. Autocomplete and Tooltip



Experiment Details

In this, we are auto completing an input field. The field shows suggestions as soon as first charachter is entered. This is achieved by autocomplete() function.The tooltip is used to display a text when a mouse is hovered on some input field. This is done with the help of tooltip() function.


Code Snippet

Autocomplete
                  $(function () {

                    var availableTags = [

                    "ActionScript",

                    "AppleScript",

                    "Asp",

                    "BASIC",

                    "C",

                    "C++",

                    "Clojure",

                    "COBOL",

                    "ColdFusion",

                    "Erlang",

                    "Fortran",

                    "Groovy",

                    "Haskell",

                    "Java",

                    "JavaScript",

                    "Lisp",

                    "Perl",

                    "PHP",

                    "Python",

                    "Ruby",

                    "Scala",

                    "Scheme"

                  ];

                  $("#tags").autocomplete({

                    source: availableTags,

                    autoFocus: true,

                    _renderItem: function (ul, item) {

                    return $("<li>")

                    .attr("data-value", item.value)

                    .append(item.label)

                    .appendTo(ul);

                  },

                  focus: function (event, ui) {

                    $(".ui-helper-hidden-accessible").hide();

                    event.preventDefault();

                    }

                   });

                 });
Tooltip
                    $(function () {

                        $(document).tooltip();

                    });
